Generating the two summary files, one by citytown and one by zip.

SRECcity <- SREC %>% 
  ungroup() %>%
  filter(type=="Residential") %>% 
  group_by(citytown) %>%
  dplyr::summarise(systems=n(), totcapacity=sum(capacity, na.rm=TRUE), totalcost=sum(totcost, na.rm=TRUE))
save(SRECcity, file = "SRECcity_6_16.RData")

SRECzip <- SREC %>% 
  ungroup() %>%
  filter(type=="Residential") %>% 
  group_by(zip) %>%
  dplyr::summarise(systems=n(), totcapacity=sum(capacity, na.rm=TRUE), totalcost=sum(totcost, na.rm=TRUE))
save(SRECzip, file = "SRECzip_6_16.RData")
